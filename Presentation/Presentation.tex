\begin{itemize}
	\item $ a \bullet b = b \bullet a $
	\item $ a \bullet (\textbf{r}b + c) = \textbf{r}(a \bullet b) + (a \bullet c) $
	\\
	\item $ a \bullet 0 = 0 $
	\item $ a \bullet a = | a^2 |$
